\begin{ex}%[2D1H1-2]
Cho hàm số $y=f(x)=\dfrac{ax^2+bx+c}{mx+n}$ (với $a, m \ne 0$) có đồ thị là đường cong như hình.
\begin{center}
\begin{tikzpicture}[scale=0.8, font=\footnotesize, line join=round, line cap=round, >=stealth]
0	\def\xmin{-6}\def\xmax{2}\def\ymin{-6}\def\ymax{4}
	\draw[->] (\xmin-0.2,0)--(\xmax+0.2,0) node[below] {\footnotesize $x$};
	\draw[->] (0,\ymin-0.2)--(0,\ymax+0.2) node[right] {\footnotesize $y$};
	\fill (0,0)circle (1pt) node [below left] {\footnotesize $O$};
	\foreach \x in {-1}
	\fill (\x,0)circle (1pt) node [below] {$\x$};
	\foreach \x in {-3,-2}

	\foreach \y in {-3,-1,1}
	\fill (0,\y)circle (1pt) node [right] {$\y$};
	\clip (\xmin,\ymin) rectangle (\xmax,\ymax);
	\draw[dashed] (-2,\ymin)--(-2,\ymax);
	\draw[dashed,domain=\xmin:\xmax] plot (\x,{(\x)+1});
	\draw[smooth,samples=200,domain=\xmin:-2.1] plot (\x,{(((\x)^2)+3*(\x)+3)/(1*(\x)+2)});
	\draw[smooth,samples=200,domain=-1.9:\xmax] plot (\x,{(((\x)^2)+3*(\x)+3)/(1*(\x)+2)});
	\draw[dashed] (-3,0)--(-3,-3)--(0,-3)
	 (-2,0)--(-2,-1)--(0,-1)
	;\fill (-3,-7) circle (1pt);
	\draw[dashed] (-1,0)--(-1,1)--(0,1) ;\fill (-1,1) circle (1pt);
\end{tikzpicture}	
\end{center}	
	\choiceTF
	{\True Hàm số đồng biến trên các khoảng $(-\infty;-3)$ và $\left(-1;+\infty\right)$}
	{\True $f(2024)<f(2025)$}
	{Đồ thị hàm số có tiệm cận đứng $x=-3$ và đường tiệm cận xiên $y=x+1$}
	{\True Đồ thị hàm số đi qua điểm $A\left(2;\dfrac{13}{4}\right)$}
	\loigiai{
	\begin{itemchoice}
			\itemch %Đúng.\\
Dựa vào đồ thị hàm số, ta nhận thấy hàm số đồng biến trên các khoảng $(-\infty;-3)$ và $\left(-1;+\infty\right)$.			
			\itemch %Đúng.\\
			Hàm số đồng biến trên các khoảng  $\left(-1;+\infty\right)$.\\
Mà $2024$; $2025 \in \left(-1;+\infty\right)$  nên $f(2024)<f(2025)$.			
			\itemch %Sai.\\
Đồ thị hàm số có tiệm cận đứng là $x=-2$.\\
Gọi đường tiệm cận xiên có dạng $y=a'x+b'$.\\
Dựa vào đồ thị, ta thấy đường tiệm cận xiên đi qua hai điểm $(0;1)$ và $(-2;-1)$.\\
Suy ra $\heva{&b'=1\\&-2a'+b'=-1} \Leftrightarrow \heva{&b'=1\\&a'=1.}$\\
Vậy đường tiệm cận xiên là $y=x+1$.
\itemch %Đúng.\\
	Vì đồ thị hàm số có tiệm cận xiên là $y=x+1$ và tiệm cận đứng là $x=-2$ nên hàm số có dạng $y=x+1+\dfrac{k}{x+2}$.\\
	Vì đồ thị hàm số đi qua điểm $(-1;1)$ nên ta có $1=-1+1+\dfrac{k}{-1+2}\Rightarrow k=1$.\\
	Do đó hàm số đã cho là $y=x+1+\dfrac{1}{x+2}=\dfrac{x^2+3x+3}{x+2}$.\\
	Thay $x=2$ ta tính được $y=\dfrac{13}{4}$ nên đồ thị hàm số đã cho đi qua điểm $A\left(2;\dfrac{13}{4}\right)$.
		\end{itemchoice}
		}	
\end{ex}

\begin{ex}%[2D1H1-5]
	Sau khi phát hiện một bệnh dịch, các chuyên gia y tế ước tính số người nhiễm bệnh kể từ ngày xuất hiện bệnh nhân đầu tiên đến ngày thứ $t$ là $f(t)=45t^2-t^3$, $ t=0$, $1$, $2$, $\ldots$, $25$. Nếu coi $f(t)$ là hàm số xác định trên đoạn $[0; 25]$ thì đạo hàm $f'(t)$ được xem là tốc độ truyền bệnh (người/ngày) tại thời điểm $t$. Giả sử khoảng thời gian mà tốc độ truyền bệnh giảm là từ ngày thứ $m$ đến ngày thứ $n$. Khi đó $n-m$ bằng bao nhiêu?
	\loigiai{
		Tốc độ truyền bệnh $y=f'(t)=-3t^2+90t$ với $t\in [0; 25]$.\\
		Ta có $y'=-6t+90$; $y'=0\Leftrightarrow t=15$.\\
		Bảng biến thiên
		\begin{center}
			\begin{tikzpicture}[>=stealth,font=\footnotesize,scale=1]
				\tkzTabInit[nocadre=false,lgt=1.2,espcl=3,deltacl=.55]{$x$ /1, $y'$ /1, $y$ /2}{$0$, $15$, $25$}
				\tkzTabLine{, +, 0, -, }
				\tkzTabVar{-/ $0$,+/$675$,-/$375$}
			\end{tikzpicture}
		\end{center}
		Tốc độ truyền bệnh giảm từ ngày $15$ đến ngày $25$.\\
		Do đó $m=15$, $n=25$.\\ Suy ra $m-n=10$.
	}
\end{ex}
